  <script>
    document.getElementById("fruitForm").addEventListener("submit", function(event) {
      event.preventDefault();

      const fruta = document.getElementById("fruitName").value.trim();
      const preco = parseFloat(document.getElementById("price").value);
      const quantidade = parseFloat(document.getElementById("quantity").value);
      const total = preco * quantidade;

      const resultado = document.getElementById("result");

      if (fruta && !isNaN(total)) {
        resultado.innerHTML = `Total para <strong>${fruta}</strong>: R$ ${total.toFixed(2)}`;
      } else {
        resultado.textContent = "Preencha todos os campos corretamente.";
      }
    });
